获取VIP套餐列表

请求示例:

    "http://127.0.0.1:8000/api/vip/plans"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

fetch(url, {
    method: "GET",
    headers,
}).then(response => response.json());
$client = new \GuzzleHttp\Client();
$url = 'http://127.0.0.1:8000/api/vip/plans';
$response = $client->get(
    $url,
    [
        'headers' => [
            'Content-Type' => 'application/json',

    ]
);
